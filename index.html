function doGet(e) {
  var callback = e.parameter.callback;
  var result = handleRequest(e);
  if (callback) {
    return ContentService.createTextOutput(callback + "(" + result + ")")
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return ContentService.createTextOutput(result)
    .setMimeType(ContentService.MimeType.JSON);
}

function handleRequest(e) {
  try {
    var action = e.parameter.action;
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    if (action === 'test') {
      return ContentService.createTextOutput(JSON.stringify({status:'ok', message:'Connected!'}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    if (action === 'sync') {
      var data = JSON.parse(decodeURIComponent(e.parameter.data));
      writeSheet(ss, 'Sales', formatSalesData(data));
      writeSheet(ss, 'Expenses', formatExpenseData(data));
      writeSheet(ss, 'Stock', formatStockData(data));
      writeSheet(ss, 'Subscribers', formatSubData(data));
      writeSheet(ss, 'PurchaseLists', formatPurchaseData(data));
      var logSheet = getOrCreateSheet(ss, 'SyncLog');
      logSheet.appendRow([new Date().toLocaleString('en-IN'), 'Sync', data.currentWeek || '']);
      return ContentService.createTextOutput(JSON.stringify({status:'ok', synced: new Date().toLocaleString('en-IN')}))
        .setMimeType(ContentService.MimeType.JSON);
    }

    return ContentService.createTextOutput(JSON.stringify({status:'error', message:'Unknown action'}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService.createTextOutput(JSON.stringify({status:'error', message: err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

function getOrCreateSheet(ss, name) {
  return ss.getSheetByName(name) || ss.insertSheet(name);
}

function writeSheet(ss, name, rows) {
  var sheet = getOrCreateSheet(ss, name);
  sheet.clearContents();
  if (rows.length > 0) {
    sheet.getRange(1, 1, rows.length, rows[0].length).setValues(rows);
  }
}

function formatSalesData(data) {
  var rows = [['Week','Day','Full Thali','Dal Rice','Chap Bhaji','Chapati','Rice','Dal','Bhaji','Buttermilk','Sweet','Papad','Thali Revenue','Bowls Sold','Bowls Free','Bowl Revenue','Total Revenue']];
  var sales = data.sales || {};
  Object.keys(sales).forEach(function(week) {
    Object.keys(sales[week]).forEach(function(day) {
      var s = sales[week][day];
      var t = s.thali || {};
      var b = s.bowl || {};
      var tRev = (t.full_thali||0)*70+(t.dal_rice||0)*40+(t.chap_bhaji||0)*45+(t.chapati||0)*10+(t.rice||0)*20+(t.dal||0)*15+(t.bhaji||0)*20+(t.buttermilk||0)*25+(t.sweet||0)*35+(t.papad||0)*10;
      var bRev = (b.sold||0)*280;
      rows.push([week,day,t.full_thali||0,t.dal_rice||0,t.chap_bhaji||0,t.chapati||0,t.rice||0,t.dal||0,t.bhaji||0,t.buttermilk||0,t.sweet||0,t.papad||0,tRev,b.sold||0,b.free||0,bRev,tRev+bRev]);
    });
  });
  return rows;
}

function formatExpenseData(data) {
  var rows = [['Week','Day','Service','Item Name','Qty','Unit','Amount','Vendor','Notes']];
  var exp = data.expenses || {};
  Object.keys(exp).forEach(function(week) {
    Object.keys(exp[week]).forEach(function(day) {
      var e = exp[week][day];
      var thaliP = (e.thali_purchases || []).map(function(p) { return Object.assign({}, p, {svc:'Thali'}); });
      var bowlP = (e.bowl_purchases || []).map(function(p) { return Object.assign({}, p, {svc:'Bowl'}); });
      thaliP.concat(bowlP).forEach(function(p) {
        rows.push([week,day,p.svc,p.name||'',p.qty||'',p.unit||'',p.amount||0,p.vendor||'',p.notes||'']);
      });
    });
  });
  return rows;
}

function formatStockData(data) {
  var rows = [['ID','Name','Qty','Unit','Min Alert','Category','Service']];
  (data.stock || []).forEach(function(s) {
    rows.push([s.id,s.name,s.qty,s.unit,s.min,s.category,s.service]);
  });
  return rows;
}

function formatSubData(data) {
  var rows = [['ID','Name','Phone','Service','Plan','Meals/Day','Start','End','Price','Payment','Building','Floor','Landmark','Address','Delivered','Skipped','Extra Days']];
  (data.subscribers || []).forEach(function(s) {
    var delivered = (s.deliveryLog||[]).filter(function(l){return l.status==='delivered';}).length;
    var skipped = (s.deliveryLog||[]).filter(function(l){return l.status==='skipped';}).length;
    rows.push([s.id,s.name,s.phone,s.service,s.plan,s.meals,s.startDate,s.endDate,s.price||0,s.payment,s.building||'',s.floor||'',s.landmark||'',s.address||'',delivered,skipped,s.extraDays||0]);
  });
  return rows;
}

function formatPurchaseData(data) {
  var rows = [['Week','Day','Item','Marathi Name','Qty','Unit','Est Price','Type']];
  var purchases = data.purchases || {};
  Object.keys(purchases).forEach(function(week) {
    Object.keys(purchases[week]).forEach(function(day) {
      var p = purchases[week][day];
      (p.bowlIngredients||[]).filter(function(i){return i.selected;}).forEach(function(i) {
        rows.push([week,day,i.name,'',i.buyQty||'',i.buyUnit||'','','Bowl']);
      });
      (p.thaliItems||[]).forEach(function(i) {
        rows.push([week,day,i.name,i.nameMr||'',i.qty||'',i.unit||'',i.estPrice||0,'Thali']);
      });
      (p.otherItems||[]).forEach(function(i) {
        rows.push([week,day,i.name,i.nameMr||'',i.qty||'',i.unit||'',i.estPrice||0,'Other']);
      });
    });
  });
  return rows;
}
